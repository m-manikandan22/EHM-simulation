def get_downstream_nodes(self, start):
    visited = set()
    stack = [start]

    while stack:
        node = stack.pop()

        for _, child, data in self.graph.out_edges(node, data=True):
            if not data.get("active", True):
                continue

            if child not in visited:
                visited.add(child)
                stack.append(child)

    return visited

function drawFlow(ctx, edge, t) {
  if (!edge.active || edge.flow < 0.01) return;

  const speed = 0.002 * edge.flow;
  const progress = (t * speed) % 1;

  const x = edge.x1 + (edge.x2 - edge.x1) * progress;
  const y = edge.y1 + (edge.y2 - edge.y1) * progress;

  ctx.beginPath();
  ctx.moveTo(edge.x1, edge.y1);
  ctx.lineTo(x, y);

  ctx.strokeStyle =
    edge.source_type === "solar" ? "#FFD700" :
    edge.source_type === "battery" ? "#FF00FF" :
    edge.source_type === "wind" ? "#00BFFF" :
    "#00FFCC";

  ctx.lineWidth = 3;
  ctx.stroke();
}
