🔥 STEP 3 — ADD FLOW ANIMATION FUNCTION

function animateFlow(path) {
    if (!path || path.length < 2) return;

    let progress = (Date.now() % 2000) / 2000; // loop 2 sec

    for (let i = 0; i < path.length - 1; i++) {
        const from = gridData.nodes[path[i]];
        const to = gridData.nodes[path[i + 1]];

        const x = from.x + (to.x - from.x) * progress;
        const y = from.y + (to.y - from.y) * progress;

        ctx.beginPath();
        ctx.arc(x, y, 4, 0, Math.PI * 2);
        ctx.fillStyle = "#00f"; // blue flow
        ctx.fill();
    }
}
