```python
def get_downstream_nodes(self, start):
    visited = set()
    stack = [start]
    while stack:
        node = stack.pop()
        for _, child, data in self.graph.out_edges(node, data=True):
            if not data.get("active", True): continue
            if child not in visited:
                visited.add(child)
                stack.append(child)
    return visited
```

**Code for the #1 most visible fix (continuous directional flow)** — put this in your `GridGraph` component:

```tsx
// frontend/src/components/GridGraph.tsx
const drawFlow = (ctx: CanvasRenderingContext2D, edge: any, t: number) => {
  if (!edge.active || edge.flow < 0.01) return;
  const speed = 0.003 * edge.flow;
  const progress = (t * speed) % 1;

  const x = edge.x1 + (edge.x2 - edge.x1) * progress;
  const y = edge.y1 + (edge.y2 - edge.y1) * progress;

  ctx.beginPath();
  ctx.moveTo(edge.x1, edge.y1);
  ctx.lineTo(x, y);

  // color by source
  const color = edge.source_type === "solar" ? "#FFD700" :
                edge.source_type === "battery" ? "#FF00FF" :
                edge.source_type === "wind" ? "#00BFFF" : "#00FFCC";

  ctx.strokeStyle = color;
  ctx.lineWidth = 4;
  ctx.shadowBlur = 15;
  ctx.shadowColor = color;
  ctx.stroke();
  ctx.shadowBlur = 0;
};
```

Run `animate()` with `requestAnimationFrame` using current `gridState`.
