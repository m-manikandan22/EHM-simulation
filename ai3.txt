```python
# ✅ FIXED
def get_downstream_nodes(self, start):
    """
    Returns ONLY nodes that receive power FROM start.
    Uses outgoing edges only (respects power flow direction).
    """
    visited = set()
    stack = [start]
    
    while stack:
        node = stack.pop()
        # ONLY follow outgoing edges (parent → child)
        for _, child, data in self.graph.out_edges(node, data=True):
            # Skip inactive edges (broken cables or open switches)
            if not data.get("active", True):
                continue
            if child not in visited:
                visited.add(child)
                stack.append(child)
    
    return visited

def _validate_tie_candidate(self, u, v, grid, cluster, powered):
    """Validate that a tie candidate is a REAL switch."""
    edge_data = grid.graph.get_edge_data(u, v, default={})
    
    # Must be a tie switch
    if not edge_data.get("is_tie_switch", False):
        return False, "Not a tie switch"
    
    # Must be normally open (inactive)
    if edge_data.get("active", False):
        return False, "Tie already closed"
    
    # Exactly one end in cluster, other powered
    u_in = u in cluster
    v_in = v in cluster
    if not ((u_in and v in powered) or (v_in and u in powered)):
        return False, "Invalid connection"
    
    return True, "Valid"
```

```python
def _dispatch_control_signal(self, action_name: str, state_before: dict, grid: SmartGrid, ems=None):
    """Execute the control signal with proper validation."""
    
    if action_name == "increase_generation":
        # Find all generators and ramp up
        ramped = 0
        for nid, node in grid.nodes.items():
            if node.node_type in ["generator_solar", "generator_wind", "generator_coal"]:
                old_gen = node.generation
                node.generation = min(node.generation * 1.2, node.max_generation if hasattr(node, 'max_generation') else 15.0)
                ramped += 1
        return f"⚡ Increased generation on {ramped} generators"
    
    elif action_name == "use_battery":
        # Find batteries with charge
        discharged = 0
        for nid, node in grid.nodes.items():
            if node.node_type == "battery" and node.battery_level > 0.3:
                discharge = min(2.0, node.battery_level * node.battery_capacity * 0.2)
                node.generation += discharge
                node.battery_level -= discharge / node.battery_capacity
                discharged += 1
        return f"🔋 Discharged {discharged} batteries"
    
    elif action_name == "use_supercapacitor":
        discharged = 0
        for nid, node in grid.nodes.items():
            if node.node_type == "supercap" and node.supercap_level > 0.2:
                discharge = min(0.5, node.supercap_level * node.supercap_capacity)
                node.generation += discharge
                node.supercap_level -= discharge / node.supercap_capacity
                discharged += 1
        return f"⚡ Supercapacitor discharge on {discharged} nodes"
    
    elif action_name == "shift_load":
        # Shed non-critical load
        shed_mw = 0
        for nid, node in grid.nodes.items():
            if node.priority >= 2 and node.load > 0.1:  # Non-critical
                shed = node.load * 0.15
                node.load -= shed
                shed_mw += shed
        return f"📊 Load shifted: {shed_mw:.2f} MW deferred"
    
    elif action_name == "reroute_energy":
        return self._flisr_restore(grid, ems)
    
    return f"Unknown action: {action_name}"
```
